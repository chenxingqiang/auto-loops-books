\notationsection{LLM Inference and Serving}{%
$L$ & Prompt or context length (tokens) in prefill or cached history\\
$m$ & Number of autoregressive output tokens generated after prefill\\
$\text{BS}$ & Batch size (requests or sequences scheduled together)\\
$\text{SL}$ & Sequence length in a benchmark configuration (prompt + generated tokens)\\
$\text{KV}$ & Key--value cache tensors reused across decode steps\\
\textit{prefill} & Parallel forward over prompt tokens; bulk KV write\\
\textit{decode} & One (or few) new tokens per forward; KV read and append\\
}

\notationsection{Performance Metrics}{%
$\toksec$ & Throughput: tokens processed per second\\
$\mstok$ & Latency: milliseconds per generated token\\
$I$ & Arithmetic intensity (FLOPs moved per byte transferred)\\
$\HDBI$ & Host--Device Balance Index; $\HDBI \to 0$ indicates host-bound serving\\
$\launchpt$ & GPU (or device) kernel launches counted per output token\\
$\bytespt$ & Device memory bytes read/written per output token (weights, KV, activations)\\
}
\index{Throughput}
\index{Latency}
\index{Arithmetic intensity}

\notationsection{Compiler, GPU, and Hardware}{%
$\mathcal{IR}$ & Compiler intermediate representation before lowering\\
\textit{pass} & Analysis or transform applied to $\mathcal{IR}$ (fusion, tiling, layout)\\
\textit{tile} & Static sub-block of a tensor mapped to on-chip memory\\
$\text{SM}$ & Streaming multiprocessor on NVIDIA GPUs\\
$\text{CTA}$ & Cooperative thread array (CUDA thread block)\\
$\text{TMA}$ & Tensor Memory Accelerator (async bulk copy on Hopper/Blackwell)\\
$\text{HBM}$ & High-bandwidth memory for weights, KV cache, and spill traffic\\
$\text{AIE}$ & AMD AI Engine tile array (XDNA spatial dataflow)\\
}
\index{Tensor Memory Accelerator|see {TMA}}
\index{Intermediate representation}
